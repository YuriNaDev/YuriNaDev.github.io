# Solution

```py
import heapq


def solution(n, works):
    answer = 0
    heap = []
    for w in works:
        heapq.heappush(heap, (-w, w))

    while n > 0:
        x = heapq.heappop(heap)
        if x[1] == 0:
            break
        n -= 1
        v = x[1] - 1
        heapq.heappush(heap, (-v, v))

    for x in heap:
        answer += x[1] * x[1]
    return answer


print(solution(4, [4, 3, 3]))  # 12
print(solution(1, [2, 1, 2]))  # 6
print(solution(3, [1, 1]))  # 0
```
