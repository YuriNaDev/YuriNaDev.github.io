# Solution

```py
from math import *


def solution(n):
    answer = 0
    x = sqrt(n)
    for i in range(1, ceil(x)):
        if n % i == 0:
            answer += (i + n // i)
    if x == int(x):
        answer += int(x)
    return answer


print(solution(16))  # 31
print(solution(12))  # 28
print(solution(5))  # 6
```
