# Solution

```py
from math import gcd


def lcm(a, b):
    return a * b // gcd(a, b)


def solution(arr):
    if len(arr) == 1:
        return arr[0]
    answer = arr[0]
    for i in range(1, len(arr)):
        answer = lcm(answer, arr[i])
    return answer


print(solution([2, 6, 8, 14]))  # 168
print(solution([1, 2, 3]))  # 6
```
