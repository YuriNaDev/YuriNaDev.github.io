# Solution

```py
def solution(number, k):
    stack = []
    for n in number:
        # 뺄 수 있는 수가 남았고, 스택에 있는 수가 현재 수보다 작으면 빼버린다
        while k > 0 and stack and stack[-1] < n:
            k -= 1
            stack.pop()
        stack.append(n)
    while k > 0:
        stack.pop()
        k -= 1
    return ''.join(stack)


print(solution('1924', 2))  # 94
print(solution('1231234', 3))  # 3234
print(solution('4177252841', 4))  # 775841
```
