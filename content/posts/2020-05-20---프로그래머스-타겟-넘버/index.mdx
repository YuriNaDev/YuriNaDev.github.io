# Solution

```py
def solution(numbers, target):
    return dfs(numbers, 0, 0, target)


def dfs(numbers, val, curr_idx, target):
    if curr_idx == len(numbers):
        return 1 if val == target else 0
    left = dfs(numbers, val + numbers[curr_idx], curr_idx + 1, target)
    right = dfs(numbers, val - numbers[curr_idx], curr_idx + 1, target)
    return left + right


print(solution([1, 1, 1, 1, 1], 3))  # 5
```
