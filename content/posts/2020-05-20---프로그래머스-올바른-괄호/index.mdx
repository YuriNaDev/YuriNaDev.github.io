# Solution

```py
def solution(s):
    stack = []
    for x in s:
        if x == '(':
            stack.append(x)
        else:
            if not stack or stack[-1] == ')':
                return False
            stack.pop()
    return False if stack else True


print(solution('()()'))  # true
print(solution('(())()'))  # true
print(solution(')()('))  # false
print(solution('(()('))  # false
```
