# Solution

```py
def solution(n, times):
    l, r = 0, max(times) * n
    init, answer = r, r
    while l <= r:
        mid = (l + r) // 2  # 이분탐색으로 가정한 총 시간
        p = 0  # 처리할 수 있는 총 사람 수
        for x in times:
            p += mid // x
        if p == n:
            answer = min(answer, mid)
            r = mid - 1  # 총 시간을 줄여서 또 해당값이 있는지 검색
        elif p < n:
            l = mid + 1  # mid ↑ p ↑
        else:
            r = mid - 1
    return r + 1 if answer == init else answer


print(solution(6, [7, 10]))  # 28
print(solution(6, [3, 4, 7]))  # 9
print(solution(7, [3, 4, 7]))  # 12
```
